[docs]def find_commands(prefix=None):
    commands = []

    pattern = '{}-*'.format(prefix)
    paths = [Path(p) for p in os.get_exec_path()]
    for path in paths:
        for match in path.glob(pattern):
            if os.access(match, os.X_OK):
                commands.append(match)

    print(commands)
